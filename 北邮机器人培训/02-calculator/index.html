import re

def calculation(str):
    # r表示其后的字符串按原样表示，不使用转义字符
    pattern1=re.compile(r'-?\d+\.?-?\d*') #regular expression 1
    num = pattern1.findall(str)
    a=float(num[0])
    b=float(num[1])
    print(a,b)
    pattern2 = re.compile(r'[+\-\*\/]')#regular expression 2
    o = pattern2.findall(str,1)
    print(o)
    if o[0] == '+':
        return a+b
    elif o[0] =='-':
        return a-b
    elif o[0] =='*':
        return a*b
    elif o[0] =='/':
        return a/b

str = input()
ans = calculation(str)
print(ans)
