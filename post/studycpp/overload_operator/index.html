运算符重载

# include<iostream>
using namespace std;
class Fraction{

    private:

        int numerator;      //分子
        int denominator;    //分母

    public :

        //Constructor

        //No parameter

        Fraction();

        //Two parameter

        Fraction(int,int);

        //copy type

        Fraction(const Fraction &fraction1);

        //Destructor

        ~Fraction(){}

        //friend function

        friend Fraction operator /(Fraction &a, Fraction &b);

        //operator overloading

        Fraction operator *(Fraction &a);

        Fraction operator +(Fraction &a);

        void show(){
        cout<<numerator<<"\n"<<denominator<<"\n";
        }
};

//Constructor

//class_name :: class_name (parameter)

Fraction :: Fraction (){
numerator   = 1;
denominator = 1;
}

Fraction :: Fraction (int numerator, int denominator){
    this->numerator = numerator ;
    this->denominator = denominator ;
}

Fraction :: Fraction (const Fraction &fraction1){
numerator = fraction1.numerator;
denominator = fraction1.denominator;
}

//Friend Function

Fraction operator /(Fraction &a , Fraction &b){
    Fraction c;
    c.numerator= a.numerator*b.denominator;
    c.denominator = a.denominator*b.numerator;
    return c;
}

//Overload the operator

Fraction Fraction :: operator * (Fraction &a){
            Fraction c;
            c.numerator= a.numerator*numerator;
            c.denominator = a.denominator*denominator;
            return c;
}

Fraction Fraction :: operator + (Fraction &a){
    Fraction c;
    c.numerator= a.numerator*denominator + numerator*a.denominator;
    c.denominator = a.denominator*denominator;
    return c;

}

int main(){
Fraction fraction1(1,2);
Fraction fraction2(1,2);
Fraction m;
m= fraction1/fraction2;
cout<<"\n";
m.show();
cout<<"\n";
}
