target_link_libraries(turtle_command_server ${catkin_LIBRARIES})

编译并运行

cd ~/catkin_ws  
catkin_make  
source devel/setup.bash  
roscore  
rosrun turtlesim turtlesim_node  
rosrun learning_service turtle_command_server  
rosservice call /turtle_command "{}"
