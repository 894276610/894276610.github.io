import numpy as np
import matplotlib.pyplot as plt

#def load_data
# 定义导入数字数据的函数
def load_data(filename):
    # 打开文件
    fr = open(filename)
    # 设置存放数据的变量
    x =[]
    y =[]
    # 按行读取数据
    for line in fr.readlines():
        # 数据预处理
        #Python strip() 方法用于移除字符串头尾指定的字符（默认为空格或换行符）或字符序列
        # 去空 分割
        line = line.strip().split()
        # 提取文件内容写入变量
        x.append(float(line[0]))
        y.append(float(line[1]))
    # 转为矩阵形式方便后续插值
    xmat = np.mat(x).T
    ymat = np.mat(y).T
    # 返回两个矩阵
    return xmat, ymat

# 回归拟合
def wb_calc(xmat, ymat,lam = 0,alpha = 0.0001,maxIter = 20000):
    #设置参数
    m,n = xmat.shape
    X = np.mat(np.zeros((4,3)))
    X[:,0] = xmat
    X[:,1] = xmat.A ** 2
    X[:,2] = xmat.A ** 3

    # init W b
    W = np.mat(np.random.randn(3,1))
    b = np.mat(np.random.randn(1,1))

    W0 = W.copy()
    b0 = b.copy()
    for i in range(maxIter):
        # dw, db
        H = X*W +b
        dw = 1/m * X.T *(H-ymat) + 1/m *lam*W
        db = 1/m * np.sum(H-ymat)

        W-= alpha * dw
        b -= alpha * db

    return W0,b0,W,b

#show
xmat, ymat = load_data('raw_number')
# 输出 矩阵内容， 矩阵维度， 数据类型
# print('xmat',xmat,xmat.shape,type(xmat))
# print('ymat',ymat,ymat.shape,type(ymat))

W0,b0,W,b = wb_calc(xmat,ymat,10,0.0001,20000)

plotx = np.arange(0,7,0.001)
ploth = np.arange(0,7,0.001)

w1 = W[0,0]
w2 = W[1,0]
w3 = W[2,0]

ploth = w1*plotx + w2* plotx **2 + w3 * plotx ** 3 + b[0,0]
plt.plot(plotx,ploth,label='h_model')


# scatter 散点图
# 需要将矩阵转化为array数组形式， 设置size，color
plt.scatter(xmat.A,ymat.A,s=50,c='red',label='dataset')
# # 网格，标签设置
plt.grid(True)
plt.xlabel('x')
plt.show()
plt.ylabel('y')
