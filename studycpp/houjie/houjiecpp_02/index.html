字符串类

创建指针，需动态分配内存

#ifndef __MYSTRING__
#define __MYSTRING__

class String{
public:
    String(const char* cstr = 0);  

    //Big Three
    String(const String& str); //拷贝构造
    String& operator = (const String& str); //拷贝赋值  
    ~String(); //析构函数  

    char* get_c_str() const { return m_data;}  

private:
    char* m_data;
};

// String::function{}  

inline
String :: String(const char* cstr = 0){
    if(cstr) {
        m_data = new char[strlen(cstr) +1];
        strcpy(m_data, cstr);  
    }  
    else  {
        m_data = new char[1];
        *m_data = '\0';
    }
}

//destructor  

inline 
String:: ~String()
{
    delete[] m_data;
}

//copy constructor 

inline
String::String(const String& str){
    m_data = new char[strlen(str.m_data) +1];
    strcpy(m_data, str.m_data);
}

//copy assignment  
inline  String&
String:: operator = (const String& str){
    if(this == &str)  //这里&是取地址的意思
    return *this;  

    delete[] m_data;                                //清空原空间
    m_data = new char[ strlen(str.m_data) +1 ];     //创造空间
    strcpy(m_data, str.m_data);                     //拷贝内容
    return *this;
}

//output 函数  
#include <iostream.h>
ostream& operator <<(ostream& os, const String& str){
    os << str.get_c_str();
    return os;
}

stack heap

Stack 是存在于作用域的一块内存空间
Heap 由操作系统提供的global内存空间 程序可通过动态分配

Static local object 在作用域结束之后仍然存在，直到程序结束
global objects 作用域整个程序

heap objects

Complex c1(1,2); //stack auto object 
Complex* p = new Complex(1,3); //heap 需要手动释放空间
delete p;

new 先分配memory 再调用constructor

delete 先调用 dtor 再释放memory

array new 搭配 array delete

Complex* p = new Complex[3]; String* pa = new String[3]; delete[] pa; delete[] p;

如果涉及带有指针的类，array delete
让编译器晓得要释放几个指针指向的空间
